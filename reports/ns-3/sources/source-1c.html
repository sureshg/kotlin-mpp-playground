 class AtomicSample {
   private val _x = AtomicInt(0)
 
   val x
     get() = _x.load()
 
   fun doWork(finalValue: Int) {
     check(x == 0)
     check(_x.exchange(3) == 0)
     check(x == 3)
     check(_x.compareAndSet(3, finalValue))
   }
