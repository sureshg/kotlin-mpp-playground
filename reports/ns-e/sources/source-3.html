 val OTelExtnPlugin =
     createApplicationPlugin(name = "OTelExtnPlugin", createConfiguration = ::OTelExtnPluginConfig) {
       val onCallTimeKey = AttributeKey<Long>("onCallTimeKey")

         val onCallTime = System.currentTimeMillis()
