 fun Routing.webApp(remotePath: String, basePackage: String) {
   staticResources(remotePath, basePackage) {
     exclude { it.path.endsWith(".log") }
     default("index.html")
     modify { url, call ->
       when (call.request.path()) {
         remotePath -> call.respondRedirect("$remotePath/")
       }
     }
   }
 }
